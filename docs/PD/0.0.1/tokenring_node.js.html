var net = require('net'),
    actorify = require('actorify');

/**
 * Function Object para instanciar objetos llamados Near.
 * Un objeto Near actua como una interfaz adaptador, para que un nodo pueda comunicarse con su nodo vecino (por eso se llama Near).
 * @param {int} port Puerto del nodo vecino
 */
var Near = function Near(port) {
  this.port = port;
  this.actor = undefined;
  /**
   * Metodo del objeto Near que utilizan los nodos para enviarle a su nodo vecino el maximo valor hasta el momento y que siga el calculo el vecino.
   * @param  {int} port Puerto del nodo actual para enviarle al vecino, asi sabe de donde viene el mensaje.
   * @param  {int} max  Maximo actual en la red
   */
  this.callCalc = function (port, max) {
    // si todavia no estoy conectado al nodo vecino, me conecto.
    if (!this.actor) {
      this.actor = actorify(net.connect(this.port));
    }
    // envio el maximo actual.
    this.actor.send('calc', port, max);
  };
  this.callResult = function (port, result) {
    // si todavia no estoy conectado al nodo vecino, me conecto.
    if (!this.actor) {
      this.actor = actorify(net.connect(this.port));
    }
    // envio el maximo final.
    this.actor.send('result', port, result);
  };
};

/**
 * Function Object que se exporta como modulo en Node.js y permite instanciar un objeto Node
 * @param  {array} values   Array de valores enteros para calcular el maximo.
 * @param  {int} port     Puerto de nodo actual
 * @param  {int} nearPort Puerto del nodo vecino
 */
module.exports = function Node(values, port, nearPort) {
  var self = this;
  this.port = port;
  // instancio el nodo vecino
  this.near = new Near(nearPort);
  /*
   * rounds es la cantidad de rondas de mensajes que se requieren para que todos conozcan su maximo
   * Esto es algo que se determina por logica. Si analizamos la cantidad de mensajes 
   * que se necesitan como mucho para calcular el maximo en una red anillo
   * deterrminamos que con 2 rondas todos nuestros nodos sabrian el maximo.
   */ 
  this.rounds = 0;
  this.root = false;

  /**
   * Metodo de Node que calcula el valor maximo de lista: values
   */
  this.calcMyMax = function () {
    values.forEach(function(value) {
      if (self.max == undefined) {
        self.max = value;
      } else if (value > self.max) {
        self.max = value;
      }
    });
  };

  /**
   * Metodo de Node que compara el valor actual con el que recibe del nodo vecino
   */
  this.calcMax = function getMax(eMax) {
    if (eMax > self.max) {
      self.max = eMax;
    }
  };

  // creo un server para el nodo actual y el socket que me retorna lo transformo a Actor para operar con mensajes.
  net.createServer(function(sock){
    self.actor = actorify(sock);

    self.actor.on('calc', function (port, max) {
      console.log('Recibo el valor ' + max + ' de ' + port);
      if (self.root) {
        /** 
         * Si recibo un mensaje 'calc' y soy el nodo que inicio la busqueda (root) significa que ya se completo una ronda de mensajes en la
         * red y por lo tanto ya se conoce el maximo, tengo que enviar un mensaje a mi nodo vecino para comunicarle el resultado final.
         */
        self.max = max;
        self.near.callResult(self.port, self.max);
      } else {
        /** 
         * Si recibo un mensaje 'calc' y NO soy el nodo que inicio la busqueda (root) significa que el analisis del maximo sigue en curso.
         * Por lo tanto calculo mi maximo con el nuevo valor y envio mi resultado al siguiente nodo vecino.
         */
        self.calcAndSend(max);
      }
    });
    // define una funcion callback que se ejecuta una sola vez cuando recibe el mensaje 'result', el cual retorna el resultado final.
    self.actor.once('result', function (port, result) {
      self.max = result;
      console.log('El valor maximo es: ' + self.max);
      if (self.root) {
        process.exit(1);
      } else {
        self.near.callResult(self.port, self.max);
        process.exit(1);
      }
    });
  }).listen(this.port);

  /**
   * Metodo del objeto Node que recibe un valor de su 'nodo vecino predecesor', calcula el nuevo maximo y envia el resultado a su 'nodo vecino sucesor'.
   * @param  {int} eMax Valor maximo recibido de su 'nodo vecino predecesor'
   */
  this.calcAndSend = function (eMax) {
    if (self.max == undefined) {
      self.calcMyMax();
    }
    if (eMax != undefined) {
       self.calcMax(eMax);
    }
    console.log('Envio el valor ' + self.max + ' a mi vecino ' + self.near.port);
    self.near.callCalc(self.port, self.max);
  };

  /**
   * Metodo del objeto Node que inicia la busqueda del maximo en la red.
   * El nodo que inicia la busqueda se lo conoce como root o principal y es el encargado de determinar que hacer en cada ronda de mensajes.
   * 1er ronda de mensajes es para calcular el maximo.
   * 2da ronda de mensajes comunica el resultado final y permite terminar a cada nodo.
   */
  this.start = function () {
    this.root = true;
    console.log('Inicio el filtrado en busca del maximo valor...!');
    self.calcAndSend();
  };
};
